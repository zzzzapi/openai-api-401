Python示例

from openai import OpenAI

client = OpenAI(
    base_url="https://zzzzapi.com/v1",  # 4Z API的接入点
    api_key="sk-xxxx"  # 替换成您的API密钥
)

stream = client.chat.completions.create(
    model="claude-3.5-sonnet",  # 支持多种模型

JavaScript示例

import { OpenAI } from "openai";

const openai = new OpenAI({
    baseURL: "https://zzzzapi.com/v1",
    apiKey: "sk-xxxx", // 替换成您的API密钥
});

async function main() {
    const stream = await openai.chat.completions.create({
        model: "gpt-4o",
        messages: [{ role: "user", content: "写一个React组件实现待办事项列表" }],
        stream: true,
    });
    
    for await (const chunk of stream) {
        process.stdout.write(chunk.choices[0]?.delta?.content || "");
    }
}

main();

Java示例

import com.theokanning.openai.OpenAiApi;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class OpenAiExample {
    public static void main(String[] args) {
        String apiKey = "sk-xxxx"; // 替换成您的API密钥
        String baseUrl = "https://zzzzapi.com/v1";
        
        OpenAiService service = new OpenAiService(apiKey, baseUrl);
        
        List messages = new ArrayList<>();
        messages.add(new ChatMessage("user", "用Java实现一个简单的HTTP服务器"));
        
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("mistral-large")
                .messages(messages)
                .build();
        
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
